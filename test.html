/* ===== PICKLEBALL MANAGER API ===== * Dán toàn bộ code này vào Apps Script * Deploy: Execute as "Me", Who has access "Anyone" */ const CONFIG = { SPREADSHEET_ID: "1GUr4TFhq_Ha65en9vthhtii35-RWg28qeEcLiVJaVek", SHEET_PLAYERS: "DanhSachVDV", SHEET_SCHEDULE: "LichThiDau" }; // ===== ENTRY POINTS ===== function doGet(e) { return handleRequest(e); } function doPost(e) { return handleRequest(e); } function handleRequest(e) { try { const action = e?.parameter?.action; const body = parseBody(e); Logger.log(`[API] action:${action} body:${JSON.stringify(body).slice(0,200)}`); switch(action) { case "ping": return jsonResponse({ success: true, message: "API OK", time: new Date().toISOString() }); case "getPlayers": return jsonResponse(getPlayers()); case "savePlayers": return jsonResponse(savePlayers(body?.players)); case "createSchedule": return jsonResponse(createSchedule()); default: return jsonResponse({ success: false, error: `Action không hợp lệ: ${action}` }); } } catch (err) { Logger.log(`[ERROR] ${err.message}\n${err.stack}`); return jsonResponse({ success: false, error: err.message }); } } // ===== UTILS ===== function parseBody(e) { if (!e?.postData?.contents) return {}; try { return JSON.parse(e.postData.contents); } catch { return {}; } } function jsonResponse(data) { return ContentService .createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); } function getSheet(name) { const ss = SpreadsheetApp.openById(CONFIG.SPREADSHEET_ID); const sheet = ss.getSheetByName(name); if (!sheet) throw new Error(`Sheet '${name}' không tồn tại`); return sheet; } function getOrCreateSheet(name) { const ss = SpreadsheetApp.openById(CONFIG.SPREADSHEET_ID); let sheet = ss.getSheetByName(name); if (!sheet) sheet = ss.insertSheet(name); return sheet; } // ===== PLAYER MANAGEMENT ===== function getPlayers() { const sheet = getSheet(CONFIG.SHEET_PLAYERS); const lastRow = sheet.getLastRow(); if (lastRow < 2) return { success: true, players: [] }; const values = sheet.getRange(2, 1, lastRow - 1, 1) .getValues() .flat() .map(v => String(v).trim()) .filter(v => v); return { success: true, players: values }; } function savePlayers(arr) { if (!Array.isArray(arr)) throw new Error("Dữ liệu không hợp lệ"); const sheet = getSheet(CONFIG.SHEET_PLAYERS); sheet.getRange("A2:A").clearContent(); const clean = arr.map(v => String(v).trim()).filter(v => v); if (clean.length > 0) { sheet.getRange(2, 1, clean.length, 1).setValues(clean.map(p => [p])); } return { success: true, message: `Đã lưu ${clean.length} VĐV`, count: clean.length }; } // ===== SCHEDULE GENERATION (Optimized for Web App) ===== function createSchedule() { const players = getPlayers().players; // Validate if (players.length < 5 || players.length > 12) { throw new Error(`Cần 5-12 VĐV để tạo lịch. Hiện có: ${players.length}`); } const N = players.length; const mode = N <= 6 ? "NO_REPEAT" : N <= 8 ? "HYBRID" : "FULL_MATCH"; const targetMatches = N - 1; // Mỗi VĐV nên chơi ~N-1 trận // Stats tracking const playerMatches = {}; const pairHistory = {}; const opponentHistory = {}; players.forEach(p => playerMatches[p] = 0); const pairKey = (a, b) => [a, b].sort().join("|"); const matches = []; // ⚠️ Giới hạn vòng lặp để tránh timeout (GAS max 6 phút) const maxIterations = N <= 8 ? 4000 : 6000; let iterations = 0; while (iterations++ < maxIterations) { // Check stop condition (cho mode không phải NO_REPEAT) if (mode !== "NO_REPEAT" && players.every(p => playerMatches[p] >= targetMatches)) { break; } let bestMatch = null; let bestScore = Infinity; // Duyệt tất cả combination 4 người → 2 đội for (let i = 0; i < N; i++) { for (let j = i + 1; j < N; j++) { for (let k = 0; k < N; k++) { for (let l = k + 1; l < N; l++) { const team1 = [players[i], players[j]]; const team2 = [players[k], players[l]]; const allPlayers = [...team1, ...team2]; // Phải có 4 người khác nhau if (new Set(allPlayers).size < 4) continue; const pair1Key = pairKey(team1[0], team1[1]); const pair2Key = pairKey(team2[0], team2[1]); // ===== MODE CONTROL ===== if (mode === "NO_REPEAT") { // Không cho phép lặp cặp đã chơi if (pairHistory[pair1Key] || pairHistory[pair2Key]) continue; } if (mode === "HYBRID") { // 60% chance skip nếu cặp đã chơi if ((pairHistory[pair1Key] || pairHistory[pair2Key]) && Math.random() < 0.6) continue; } // ===== SCORING (càng thấp càng tốt) ===== let score = 0; // Ưu tiên: người ít trận được chọn trước allPlayers.forEach(p => { score += (playerMatches[p] || 0) * 4; }); // Phạt: cặp đã từng chơi chung score += (pairHistory[pair1Key] || 0) * 45; score += (pairHistory[pair2Key] || 0) * 45; // Phạt: đối thủ đã gặp nhau team1.forEach(p1 => { team2.forEach(p2 => { const oppKey = pairKey(p1, p2); score += (opponentHistory[oppKey] || 0) * 12; }); }); // Random nhỏ để tránh bias score += Math.random() * 0.1; if (score < bestScore) { bestScore = score; bestMatch = { team1, team2 }; } } } } } // Không tìm được match phù hợp if (!bestMatch) break; // Lưu match matches.push(bestMatch); const { team1, team2 } = bestMatch; // Update stats [...team1, ...team2].forEach(p => { playerMatches[p] = (playerMatches[p] || 0) + 1; }); pairHistory[pairKey(team1[0], team1[1])] = (pairHistory[pairKey(team1[0], team1[1])] || 0) + 1; pairHistory[pairKey(team2[0], team2[1])] = (pairHistory[pairKey(team2[0], team2[1])] || 0) + 1; team1.forEach(p1 => { team2.forEach(p2 => { const key = pairKey(p1, p2); opponentHistory[key] = (opponentHistory[key] || 0) + 1; }); }); } // ===== WRITE TO SHEET ===== const sheet = getOrCreateSheet(CONFIG.SHEET_SCHEDULE); sheet.clearContents(); // Header sheet.appendRow(["Trận", "P1", "P2", "P3", "P4", "Điểm1", "Điểm2", "Thắng1", "Thắng2"]); sheet.getRange(1, 1, 1, 9) .setFontWeight("bold") .setBackground("#1e293b") .setFontColor("#e2e8f0") .setHorizontalAlignment("center"); // Matches with formulas matches.forEach((m, idx) => { const row = idx + 2; sheet.appendRow([ idx + 1, m.team1[0], m.team1[1], m.team2[0], m.team2[1], "", "", // Điểm để người dùng nhập `=IF(F${row}>G${row},1,0)`, // Thắng1 `=IF(G${row}>F${row},1,0)` // Thắng2 ]); }); // Auto-format sheet.autoResizeColumns(1, 9); sheet.getRange("H2:I").setHorizontalAlignment("center"); // ===== RETURN RESULT ===== const stats = {}; players.forEach(p => stats[p] = playerMatches[p] || 0); Logger.log(`[SCHEDULE] Created ${matches.length} matches | Mode: ${mode} | Players: ${N}`); return { success: true, message: `✅ Tạo thành công ${matches.length} trận`, matches: matches.length, mode: mode, players: N, stats: stats }; }
